    <script>

        // Global variables for game state

        let pegs = [[], [], []]; // Array to store disks on each peg

        let selectedDisk = null; // Stores the disk element currently selected

        let selectedPegIndex = -1; // Stores the index of the peg the disk was picked from

        let moves = 0;

        let numDisks = 3; // Default number of disks

        let minMoves = 0; // Minimum number of moves to solve


        // DOM Elements

        const pegElements = [

            document.getElementById('peg0'),

            document.getElementById('peg1'),

            document.getElementById('peg2')

        ];

        const movesSpan = document.getElementById('moves');

        const minMovesSpan = document.getElementById('minMoves'); // New: Reference to min moves span

        const numDisksSelect = document.getElementById('numDisks');

        const resetButton = document.getElementById('resetButton');

        const messageBox = document.getElementById('messageBox');

        const messageText = document.getElementById('messageText');

        const closeMessageBoxButton = document.getElementById('closeMessageBox');


        /**

         * Displays a custom message box instead of alert().

         * @param {string} message - The message to display.

         */

        function showMessageBox(message) {

            messageText.textContent = message;

            messageBox.style.display = 'flex';

        }


        /**

         * Hides the custom message box.

         */

        function hideMessageBox() {

            messageBox.style.display = 'none';

        }


        /**

         * Initializes or resets the game board with the specified number of disks.

         */

        function initializeGame() {

            // Clear existing disks from pegs

            pegElements.forEach(peg => {

                while (peg.firstChild) {

                    peg.removeChild(peg.firstChild);

                }

            });


            // Reset game state variables

            pegs = [[], [], []];

            selectedDisk = null;

            selectedPegIndex = -1;

            moves = 0;

            movesSpan.textContent = moves;


            // Calculate minimum moves

            minMoves = Math.pow(2, numDisks) - 1;

            minMovesSpan.textContent = minMoves; // New: Update min moves display


            // Create and add disks to the first peg

            for (let i = numDisks; i > 0; i--) {

                const disk = document.createElement('div');

                disk.classList.add('disk', `disk-color-${i % 10}`); // Assign distinct colors

                disk.style.width = `${40 + i * 15}px`; // Vary width for visual distinction

                disk.dataset.size = i; // Store disk size

                pegs[0].push(disk);

                // Ensures disks stack correctly (largest at bottom, smallest at top) due to column-reverse

                pegElements[0].appendChild(disk);

            }

        }


        /**

         * Handles the click event on a peg.

         * @param {number} pegIndex - The index of the clicked peg (0, 1, or 2).

         */

        function handlePegClick(pegIndex) {

            const currentPeg = pegs[pegIndex];


            if (selectedDisk === null) {

                // No disk selected, try to pick one up

                if (currentPeg.length > 0) {

                    selectedDisk = currentPeg[currentPeg.length - 1]; // Get the top disk

                    selectedPegIndex = pegIndex;

                    selectedDisk.classList.add('selected'); // Highlight the selected disk

                } else {

                    showMessageBox("This peg is empty. Please select a disk from another peg.");

                }

            } else {

                // A disk is already selected, try to place it

                if (selectedPegIndex === pegIndex) {

                    // Clicking the same peg deselects the disk

                    selectedDisk.classList.remove('selected');

                    selectedDisk = null;

                    selectedPegIndex = -1;

                } else {

                    // Attempt to move the disk

                    const diskSize = parseInt(selectedDisk.dataset.size);

                    const topDiskOnTargetPeg = currentPeg.length > 0 ? parseInt(currentPeg[currentPeg.length - 1].dataset.size) : Infinity;


                    if (diskSize < topDiskOnTargetPeg) {

                        // Valid move: smaller disk on top of a larger one

                        pegs[selectedPegIndex].pop(); // Remove from source peg

                        currentPeg.push(selectedDisk); // Add to target peg


                        // Visually move the disk

                        pegElements[selectedPegIndex].removeChild(selectedDisk);

                        // Ensures disks stack correctly (largest at bottom, smallest at top) due to column-reverse

                        pegElements[pegIndex].appendChild(selectedDisk);


                        selectedDisk.classList.remove('selected'); // Deselect the disk

                        selectedDisk = null;

                        selectedPegIndex = -1;

                        moves++;

                        movesSpan.textContent = moves;


                        // Check for win condition

                        checkWinCondition();

                    } else {

                        // Invalid move

                        showMessageBox("Invalid move! You can only place a smaller disk on top of a larger one.");

                    }

                }

            }

        }


        /**

         * Checks if the game has been won.

         */

        function checkWinCondition() {

            // Win if all disks are on the last peg (peg 2) or the middle peg (peg 1)

            if (pegs[1].length === numDisks || pegs[2].length === numDisks) {

                let winMessage = `Congratulations! You solved the Tower of Hanoi in ${moves} moves.`;

                if (moves === minMoves) {

                    winMessage += " That's the minimum number of moves!";

                } else {

                    winMessage += ` The minimum moves for ${numDisks} disks is ${minMoves}.`;

                }

                showMessageBox(winMessage);

            }

        }


        // Event Listeners

        pegElements.forEach((peg, index) => {

            peg.addEventListener('click', () => handlePegClick(index));

        });


        numDisksSelect.addEventListener('change', (event) => {

            numDisks = parseInt(event.target.value);

            initializeGame();

        });


        resetButton.addEventListener('click', initializeGame);

        closeMessageBoxButton.addEventListener('click', hideMessageBox);


        // Initialize the game when the page loads

        document.addEventListener('DOMContentLoaded', initializeGame);

    </script>
